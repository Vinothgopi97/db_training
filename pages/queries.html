PostgreSQL Queries

Basic Queries

SELECT Queries

-- Basic select
SELECT * FROM users;

-- Select specific columns
SELECT name, email FROM users;

-- Select with conditions
SELECT * FROM users WHERE age >= 18;

-- Select with sorting
SELECT * FROM users ORDER BY name ASC;

Joins

Different Types of Joins

-- Inner Join
SELECT orders.id, users.name
FROM orders
INNER JOIN users ON orders.user_id = users.id;

-- Left Join
SELECT users.name, orders.id
FROM users
LEFT JOIN orders ON users.id = orders.user_id;

-- Right Join
SELECT users.name, orders.id
FROM users
RIGHT JOIN orders ON users.id = orders.user_id;

-- Full Outer Join
SELECT users.name, orders.id
FROM users
FULL OUTER JOIN orders ON users.id = orders.user_id;

Aggregate Functions

-- Count total records
SELECT COUNT(*) FROM users;

-- Sum values
SELECT SUM(amount) FROM orders;

-- Average value
SELECT AVG(age) FROM users;

-- Group By with aggregates
SELECT category, COUNT(*)
FROM products
GROUP BY category;

Subqueries

-- Subquery in WHERE clause
SELECT name
FROM users
WHERE id IN (SELECT user_id FROM orders);

-- Subquery in FROM clause
SELECT avg_age
FROM (SELECT AVG(age) as avg_age FROM users) as subq;

Common Table Expressions (CTE)

WITH active_users AS (
SELECT * FROM users WHERE status = 'active'
)
SELECT * FROM active_users
WHERE age > 25;

Advanced Query Features

  • Window Functions: Functions that perform calculations across a set of rows
  • CASE Statements: Conditional logic in queries
  • UNION/INTERSECT: Combining results from multiple queries
  • Views: Virtual tables based on query results

Query Optimization Tips:

  • Use appropriate indexes
  • Avoid SELECT *
  • Use EXPLAIN ANALYZE to understand query performance
  • Write efficient JOIN conditions
  • Use appropriate WHERE clauses
